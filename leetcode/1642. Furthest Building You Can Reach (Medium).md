# 1642. Furthest Building You Can Reach (Medium)

heap

```python
class Solution:
    def furthestBuilding(self, h: List[int], bricks: int, ladders: int) -> int:
        lv = []

        for x in range(1, len(h)):
            gap = h[x] - h[x-1]
            if gap <= 0:
                continue
            lv.append(gap)
            if gap <= bricks:
                bricks -= gap
                continue
            elif ladders != 0:
                lv.sort()
                ladders -= 1
                bricks += lv.pop() - gap
                continue
            else:
                return x - 1

        return len(h) - 1
```
