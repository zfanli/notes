```python
class Solution:
    def scheduleCourse(self, courses: List[List[int]]) -> int:
        c = [x for x in sorted(courses, key=lambda x: x[1])]
        heap = []
        time = 0
        for i, x in enumerate(c):
            if time + x[0] <= x[1]:
                time += x[0]
            elif len(heap) > 0 and heap[-1] > x[0] and time - heap[-1] + x[0] <= x[1]:
                time +=  - heap.pop() + x[0]
            else:
                continue

            heap.append(x[0])
            heap.sort()

        return len(heap)
```
